Most used image formats of Top Alexa Websites

SELECT mimeType, ct * 100 AS percent FROM (
  SELECT COUNT(requestid) as total, 
    RATIO_TO_REPORT(total) OVER() AS ct, mimeType
  FROM [httparchive:runs.latest_requests]
  WHERE mimeType LIKE "image/%"
  GROUP BY mimeType
  ORDER BY ct DESC
)
